    Name = "Mini executor",

        local frame = script.Parent
local uis = game:GetService("UserInputService")
local ts = game:GetService("TweenService")

local addButton = frame.Parent:WaitForChild("AddButton")
local scriptBox = frame:WaitForChild("ScriptBox")
local executeButton = frame:WaitForChild("ExecuteButton")
local scriptHubButton = frame:WaitForChild("ScriptHubButton")

-- DRAGGING (Mobile + PC)
frame.InputBegan:Connect(function(input)
	if input.UserInputType == Enum.UserInputType.MouseButton1 or input.UserInputType == Enum.UserInputType.Touch then
		dragging = true
		dragStart = input.Position
		startPos = frame.Position

		input.Changed:Connect(function()
			if input.UserInputState == Enum.UserInputState.End then
				dragging = false
			end
		end)
	end
end)

uis.InputChanged:Connect(function(input)
	if dragging and (input.UserInputType == Enum.UserInputType.MouseMovement or input.UserInputType == Enum.UserInputType.Touch) then
		local delta = input.Position - dragStart
		frame.Position = UDim2.new(
			startPos.X.Scale,
			startPos.X.Offset + delta.X,
			startPos.Y.Scale,
			startPos.Y.Offset + delta.Y
		)
	end
end)

		end
	end
end)
